// Color scale: deep teal -> warm sand -> deep crimson
const SCALE = [
{ t: 0.00, c: [38, 76, 84] },
{ t: 0.25, c: [96, 138, 130] },
{ t: 0.50, c: [218, 196, 144] },
{ t: 0.75, c: [221, 116, 78] },
{ t: 1.00, c: [148, 31, 35] },
];

function colorFor(t) {
t = Math.max(0, Math.min(1, t));
for (let i = 0; i < SCALE.length - 1; i++) {
const a = SCALE[i], b = SCALE[i + 1];
if (t >= a.t && t <= b.t) {
const u = (t - a.t) / (b.t - a.t);
const r = Math.round(a.c[0] + (b.c[0] - a.c[0]) * u);
const g = Math.round(a.c[1] + (b.c[1] - a.c[1]) * u);
const bl = Math.round(a.c[2] + (b.c[2] - a.c[2]) * u);
return `rgb(${r},${g},${bl})`;
}
}
return ‘rgb(148,31,35)’;
}

function categoryFor(v, min, max) {
const span = max - min;
if (v < min + span * 0.2) return ‘Very low’;
if (v < min + span * 0.4) return ‘Below avg’;
if (v < min + span * 0.6) return ‘Moderate’;
if (v < min + span * 0.8) return ‘Elevated’;
return ‘High burden’;
}
