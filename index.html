/* ---- 1. Color scale: warm-to-cool diverging, perceptually tuned --- */
// Anchors low (low burden, cool teal) -> mid (warm sand) -> high (deep red)
const SCALE_STOPS = [
{ t: 0.00, c: [ 38, 76, 84] }, // deep teal
{ t: 0.25, c: [ 96, 138, 130] }, // muted teal-green
{ t: 0.50, c: [218, 196, 144] }, // warm sand
{ t: 0.75, c: [221, 116, 78] }, // ember orange
{ t: 1.00, c: [148, 31, 35] }, // deep crimson
];

function lerp(a, b, t) { return a + (b - a) * t; }
function colorFor(t) {
t = Math.max(0, Math.min(1, t));
for (let i = 0; i < SCALE_STOPS.length - 1; i++) {
const a = SCALE_STOPS[i], b = SCALE_STOPS[i+1];
if (t >= a.t && t <= b.t) {
const u = (t - a.t) / (b.t - a.t);
const r = Math.round(lerp(a.c[0], b.c[0], u));
const g = Math.round(lerp(a.c[1], b.c[1], u));
const bl = Math.round(lerp(a.c[2], b.c[2], u));
return `rgb(${r},${g},${bl})`;
}
}
return 'rgb(148,31,35)';
}

/* ---- 4. Data fetchers ---- */
async function tryFetchPLACES() {
// CDC PLACES tract-level data — Socrata API, supports CORS.
const url = `https://data.cdc.gov/resource/cwsq-ngmh.json?stateabbr=WI&countyname=Milwaukee&measureid=MHLTH&$select=locationname,data_value&$limit=2000`;
const res = await fetch(url, { mode: 'cors' });
if (!res.ok) throw new Error('PLACES fetch failed: ' + res.status);
const json = await res.json();
// locationname is the 11-digit GEOID for tracts in PLACES
const out = json
.filter(r => r.data_value != null)
.map(r => [String(r.locationname).padStart(11, '0'), parseFloat(r.data_value)]);
if (out.length < 50) throw new Error('Suspiciously few rows: ' + out.length);
return out;
}

async function tryFetchGeometries() {
// Census TIGERweb — Milwaukee County tracts as GeoJSON.
// Layer 0 of Tracts_Blocks/MapServer = 2020 tracts.
const where = encodeURIComponent(`STATE='${MILWAUKEE_FIPS_STATE}' AND COUNTY='${MILWAUKEE_FIPS_COUNTY}'`);
const url = `https://tigerweb.geo.census.gov/arcgis/rest/services/TIGERweb/Tracts_Blocks/MapServer/0/query?where=${where}&outFields=GEOID,NAME,STATE,COUNTY,TRACT&returnGeometry=true&f=geojson`;
const res = await fetch(url);
if (!res.ok) throw new Error('TIGER fetch failed: ' + res.status);
const gj = await res.json();
if (!gj.features || gj.features.length < 50) throw new Error('Bad GeoJSON');
return gj;
}

/* Synthesize fallback geometries: small grid of squares laid over
Milwaukee's bounding box, only used if TIGER fetch fails. We
colour these by the fallback values to keep the map meaningful. */
function buildFallbackGeometry(tractValues) {
// Milwaukee County rough bounds
const N = 43.19, S = 42.84, W = -88.07, E = -87.83;
const cols = 10, rows = Math.ceil(tractValues.length / cols);
const dx = (E - W) / cols;
const dy = (N - S) / rows;
const features = tractValues.map(([gid, v], i) => {
const r = Math.floor(i / cols), c = i % cols;
const x0 = W + c * dx, x1 = x0 + dx * 0.93;
const y1 = N - r * dy, y0 = y1 - dy * 0.93;
return {
type: 'Feature',
properties: { GEOID: gid, NAME: `Tract ${gid.slice(-6)}` },
geometry: { type: 'Polygon', coordinates: [[[x0,y0],[x1,y0],[x1,y1],[x0,y1],[x0,y0]]] }
};
});
return { type: 'FeatureCollection', features };
}

/* ---- 6. Main render pipeline ---- */
let geoLayer = null;

function categoryFor(v, min, max) {
const mid = (min + max) / 2;
if (v < min + (max - min) * 0.2) return 'Very low burden';
if (v < min + (max - min) * 0.4) return 'Below average';
if (v < min + (max - min) * 0.6) return 'Moderate';
if (v < min + (max - min) * 0.8) return 'Elevated';
return 'High burden';
}

function renderMap(geojson, valueMap, sourceLabel) {
const values = [...valueMap.values()];
const min = Math.min(...values);
const max = Math.max(...values);
const mean = values.reduce((a,b) => a+b, 0) / values.length;

// Update side stats
document.getElementById('statTracts').textContent = values.length;
document.getElementById('statMean').firstChild.nodeValue = mean.toFixed(1);
document.getElementById('statMax').firstChild.nodeValue = max.toFixed(1);
document.getElementById('statMin').firstChild.nodeValue = min.toFixed(1);

// Legend ticks
document.getElementById('tickMin').textContent = min.toFixed(0) + '%';
document.getElementById('tickMid').textContent = ((min + max) / 2).toFixed(0) + '%';
document.getElementById('tickMax').textContent = max.toFixed(0) + '%';

// Top 10 list
const ranked = [...valueMap.entries()]
.sort((a, b) => b[1] - a[1])
.slice(0, 10);
const topList = document.getElementById('topList');
topList.innerHTML = ranked.map(([gid, v], i) => {
const tractCode = gid.slice(-6);
return `<li data-geoid="${gid}">
<span class="rank">${String(i+1).padStart(2,'0')}</span>
<span class="name">Tract ${tractCode.replace(/^0+/, '') || '0'}<span class="sub">GEOID ${gid}</span></span>
<span class="val">${v.toFixed(1)}%</span>
</li>`;
}).join('');

// Source pill
const pill = document.getElementById('dataSourcePill');
document.getElementById('srcLabel').textContent = sourceLabel;
pill.classList.remove('live', 'fallback');
pill.classList.add(sourceLabel.includes('Live') ? 'live' : 'fallback');

// Build choropleth layer
if (geoLayer) map.removeLayer(geoLayer);
geoLayer = L.geoJSON(geojson, {
style: feat => {
const v = valueMap.get(feat.properties.GEOID);
if (v == null) return { fillOpacity: 0, color: 'transparent', weight: 0 };
const t = (v - min) / (max - min || 1);
return {
fillColor: colorFor(t),
fillOpacity: 0.86,
color: '#0d0a08',
weight: 0.6,
opacity: 0.9,
};
},
onEachFeature: (feat, layer) => {
const gid = feat.properties.GEOID;
const v = valueMap.get(gid);
if (v == null) return;
const tractCode = gid.slice(-6);
const cat = categoryFor(v, min, max);
const tipHtml = `
<div class="tip-id">TRACT ${tractCode}</div>
<div class="tip-val">${v.toFixed(1)}<span class="pct">%</span></div>
<div style="font-size:10.5px;color:var(--ink-faint);margin-top:4px;">adults with ≥14 poor mental-health days/mo</div>
<div class="tip-cat">${cat}</div>
`;
layer.bindTooltip(tipHtml, {
sticky: true,
className: 'tract-tip',
direction: 'top',
offset: [0, -4],
});
layer.on('mouseover', e => {
e.target.setStyle({ weight: 2.2, color: '#f4ede1', opacity: 1 });
e.target.bringToFront();
document.getElementById('hoverHint').innerHTML =
`<span class="dot" style="background:${colorFor((v-min)/(max-min||1))}"></span>Tract ${tractCode} · ${v.toFixed(1)}%`;
});
layer.on('mouseout', e => {
geoLayer.resetStyle(e.target);
document.getElementById('hoverHint').textContent = 'Hover a tract';
});
layer.on('click', e => {
map.fitBounds(e.target.getBounds(), { padding: [40, 40], maxZoom: 14 });
});
},
}).addTo(map);

// Fit bounds with a touch of padding
try {
map.fitBounds(geoLayer.getBounds(), { padding: [20, 20] });
} catch (e) { /* ignore */ }

// Wire up sidebar list -> map
topList.querySelectorAll('li').forEach(li => {
li.addEventListener('click', () => {
const gid = li.dataset.geoid;
geoLayer.eachLayer(l => {
if (l.feature.properties.GEOID === gid) {
map.fitBounds(l.getBounds(), { padding: [80, 80], maxZoom: 14 });
l.openTooltip();
}
});
});
});

// Hide loader
document.getElementById('loader').classList.add('hide');
}

/* ---- 7. Boot sequence ---- */
async function boot() {
const loader = document.getElementById('loader');
const loaderText = document.getElementById('loaderText');

let valueRows = null;
let geo = null;
let sourceLabel = '';
let liveData = false, liveGeo = false;

loaderText.textContent = 'Fetching CDC PLACES data…';
try {
valueRows = await tryFetchPLACES();
liveData = true;
} catch (err) {
console.warn('PLACES fetch failed, using fallback:', err);
valueRows = FALLBACK_TRACTS;
}

loaderText.textContent = 'Fetching tract geometries…';
try {
geo = await tryFetchGeometries();
liveGeo = true;
} catch (err) {
console.warn('TIGER fetch failed, using fallback grid:', err);
geo = buildFallbackGeometry(valueRows);
}

// If we got live geometry but only fallback values, restrict to GEOIDs
// we have data for — and if live data covers tracts we don't have geometry for,
// that's fine, we just won't render them.
const valueMap = new Map(valueRows);

// If we ended up with the synthetic grid, valueRows IS the geometry's authority,
// so the map matches.

if (liveData && liveGeo) sourceLabel = 'Live · CDC PLACES + TIGER';
else if (liveData) sourceLabel = 'Live data · synthetic geometry';
else if (liveGeo) sourceLabel = 'Fallback data · live geometry';
else sourceLabel = 'Demo · representative dataset';

renderMap(geo, valueMap, sourceLabel);
}

boot().catch(err => showFatal('Map render failed', (err && (err.stack || err.message)) || err));
</script>
